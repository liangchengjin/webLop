  • 一、第一种 双for循环数组去重

    1.不推荐这样使用，也不是我们想要的。

    
var ary = [1,2,1,2,1,2,3,4,2,3,4,1,2,3,4,3,2,1,4,5,2,3];		
// 当前和后一项进行对比，如果相同就删除后一项
for(var i = 0; i < ary.length-1; i++){
	var cur = ary[i]; // 第一次i=0;  
	console.log(cur);
	// i=0 第一次循环 开始比较的索引是1到末尾
	// i=1 第二次循环 开始比较的索引是2到末尾
	// 开始比较的索引是 i+1
	// 通过splice删除，原有的数组改变了，删除的那个索引以后的所有项的索引都往前提了一位
	for(var j =i+1; j < ary.length; j++){  
		// j=1 2 1 == 2 进行比较
		// j=2 1 1 == 1 进行比较，把索引2这项删掉 j-- j=1
		// j=2 2 1==2
		if(cur == ary[j]){
			ary.splice(j,1);
			j--;
		}
	}
}
console.log(ary);

/*		
Array[5]
	0: 1
	1: 2
	2: 3
	3: 4
	4: 5
	length: 5
	__proto__: Array[0]

*/

    1.数组去重用对象的方式 推荐使用

    
		var ary = [1,2,45,,23,68,79,55,2,1,2,3,4,2,3,4,1,2,3,4,3,2,1,4,5,2,3];
		var obj = {};
		for(var i = 0; i < ary.length; i++){
			var cur = ary[i];
			if(obj[cur] == cur){
				ary.splice(i, 1);
				i--;
				continue;
			}
			obj[cur] = cur;
		}
		obj = null;
		console.log(ary);  // [1, 2, 45, 23, 68, 79, 55, 3, 4, 5]<、mark>
