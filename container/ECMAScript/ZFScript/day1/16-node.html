  • 一、综合案例-选项卡-注释版

    第1步：分析需示和思路，选项卡的思路非常简单：

    /*1.鼠标滑到某一个li上面，让当前的li和对应的div有select的样式，然后再让当前的有这个select样式*/

    第2步：要操作谁就先获取谁

    
var oTab = document.getElementById("tab");
var oLis = oTab.getElementsByTagName("li");
var oDivs = oTab.getElementsByTagName("div");

    第3步：制定一个功能方法，实现我们的需求

    /*nIndex是我们定义的形参，代表当前被选中元素的索引*/

    
function tabChange(nIndex){
	for(var i = 0; i < oLis.length; i++){
		oLis[i].className = "";
		oDivs[i].className = "";
	}			
	oLis[nIndex].className = "select";
	oDivs[nIndex].className = "select";
}

    第4步：给我们的li绑定鼠标移上去的事件

    
for(var i = 0; i < oLis.length; i++){
	// 自定义属性
	oLis[i].cc = i;			// 在每一次循环的时候，把每一个li的索引定义在自己的自定义属性cc上
	oLis[i].onmouseover = function(){				
		tabChange(this.cc);		// this.cc 就是自己的属性
	}
}

  • 二、综合案例-选项卡-不加注释版

    1.不加注释版

    
var oTab = document.getElementById("tab");
var oLis = oTab.getElementsByTagName("li");
var oDivs = oTab.getElementsByTagName("div");

function tabChange(n){
	for(var i = 0; i < oLis.length; i++){
		oLis[i].className = "";
		oDivs[i].className = "";
	}
	oLis[n].className = "select";
	oDivs[n].className = "select";
}

for(var i = 0; i < oLis.length; i++){
	oLis[i].cc = i;
	oLis[i].onmouseover = function(){
		tabChange(this.cc);
	}
}
