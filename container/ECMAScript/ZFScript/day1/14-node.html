	var oList = document.getElementById("list");

	for(var i = 0; i < oLis.length; i++){
		var oLi = oLis[i];

		//oLi.cc = "chengcheng";
		
		// 鼠标移上去 添加事件
		oLi.onmouseover = function(){
			this.className = "c3"
		};
		
		// 鼠标离开
		oLi.onmouseout = function(){
			// 源来是啥就是啥
			var oidVal = this.cc;

		}			
	}
</script>

  • 三、原理

    -1.原理：操作所有的li，让按照奇偶行的规律，改变它的class样式的属性值，奇数行是 c1, 偶数行是c2

  • 四、通过元素的标签名获取一组元素

    -1.document.getElementsByTagName("元素的标签名");

    在整个文档中(获取的范围，上下文),我们通过元素的标签名(div、li、p...)的标签名来获取一组元素

    1).在获取整个文档中所有的li，它是一个集合, 我们把这个集合叫做类数组(类似于数组), 并且类数组是对象数据类型的

    
 var oLis = document.getElementsByTagName("li");
console.log(oLis);			// HTMLCollection[33]
console.log(typeof oLis);		// object

    .2).它的里面有索引从0开始的

    索引的概念：就是代表当前是第几个元素的位置下标，索引是从0开始的，索引为0代表第一个li， 索引为1代表第二个li....索引为n代表第n+1个li

    索引里面有一个length: 就是代表获取的集合的长度，或者是当前获取了多少个li

    
console.log(oLis.length);		// 33
console.log(oLis["length"]);		// 33

    3).我们想获取第一个li里的内容 

    

console.log(oLis[0]); // 获取某一个指定的元素oLis[索引]    <li class="c1">《飞鸟与鱼》（据称并非泰戈尔所著，而只是署名）</li>
console.log(oLis[1]);

    4).通过DOM方法获取的类数组还可以用.item(索引)来获取某一个

    
console.log(oLis.item(0));	//  <li class="c1">《飞鸟与鱼》（据称并非泰戈尔所著，而只是署名）</li>

  • 五、通过id来获取元素名，再通过元素的id来获取它的类名

    
var oList = document.getElementById("list");
var oLis = oList.getElementsByTagName("li");		
for(var i = 0; i < oLis.length; i++){
	// i=0; 第一次循环，我们要获取第一个li 就用 oLis[0]
	// i=1; 第二次循环，我们要获取第二个li 就用 oLis[1]
	var oLi = oLis[i]; // 每一次循环的时候在数组中把对应的li取出来
	// i=0; 偶数 第一行 是奇数行
	// i=1; 奇数 第二行 是偶数行
	// i的值是偶数代表奇数行， i的值是奇数代表偶数行
	// i%2 i除以2 取余数 如：13%2 取余数 为1
	i%2 === 0? oLi.className = "c1" : oLi.className = "c2";

  • 七.没有注释的 三行隔行变色 用switch case 来操作

    
var oList = document.getElementById("list");
var oLis = oList.getElementsByTagName("li");
for(var i = 0; i < oLis.length; i++){
	var oLi = oLis[i];
	switch(i%3){
		case 0:
			oLi.className = "c1";
			break;
		case 1:
			oLi.className = "c2";
			break
		default:
			oLi.className = "c3";			}
}
