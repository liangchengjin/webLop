    1.注释版+switch case

    
	  var oTab = document.getElementById("tab");
	  var str = "";
	  str+="<ul>";
		for(var i = 1; i <= 9; i++){ // 这是行
			var val = "c1";
			switch(i%3){
				case 0:
					val = "c1";
					break;
				case 1:
					val = "c2";
					break;
				default:
					val = "c3";
			}

			str+="<li class='"+val+"'>";
				for(var j = 1; j <= i; j++){  // 这是列
					str+="</span>";
						str+=""+j+" * "+i+" = "+(j*i)+"";
					str+="</span>";
				}
			str+="</li>";
		}
	  str+="</ul>";

	  oTab.innerHTML = str;

	  // innerHTML: 向指定元素中添加内容(可以是文字，也可以是html标签), innerHTML可以识别html标签的

	  // innerText:不能识别html标签
